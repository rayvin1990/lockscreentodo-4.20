#### 1. **前端按钮点击事件**

```javascript
// JavaScript
document.getElementById('convertButton').addEventListener('click', async function() {
  // 获取上传的图像文件（你可以从文件输入框获取）
  const imageUrl = "上传图片的URL或Base64格式";  // 替换为实际图片URL或Base64格式

  // 创建 API 请求的负载
  const payload = {
    workflow_id: "7557196778741917219",  // 你的工作流ID

      ima: imageUrl,
      promptType: "// 提示词类型 例如：\"midjouny\" || \"stableDiffusion\"",
      userQuery: "请描述一下这个图"
    }
  };

  try {
    // 发起API请求
    const response = await fetch('https://api.coze.cn/v1/workflow/stream_run', {
      method: 'POST',
      headers: {
        'Authorization': 'Bearer cztei_hdZ7rrT3TIHrses657bFc8Z6uBiebc5H6hMjVWbdU5PR5gC00GK6UxvHhH', // 替换为你的API Token
        'Content-Type': 'application/json'
      },
      body: JSON.stringify(payload)
    });

    const data = await response.json();
    if (data.debug_url) {
      // 获取 debug_url 后，轮询检查任务完成情况
      checkTaskStatus(data.debug_url);
    } else {
      console.log('Error: No debug URL returned');
    }
  } catch (error) {
    console.error('Error:', error);
  }
});

async function checkTaskStatus(debugUrl) {
  try {
    const response = await fetch(debugUrl);
    const result = await response.json();
    
    if (result.status === 'done') {
      // 显示结果
      document.getElementById('generatedPrompt').innerText = result.prompt;
    } else {
      // 任务未完成，继续轮询
      setTimeout(() => checkTaskStatus(debugUrl), 5000);  // 每5秒检查一次
    }
  } catch (error) {
    console.error('Error checking task status:', error);
  }
}
```

#### 2. **解释代码**

* **事件监听器**：当用户点击 **Convert Now** 按钮时，页面会向 Coze API 发送 POST 请求，上传图像并附带相应的参数。
* **API 请求**：`fetch` 函数发送 POST 请求到 Coze 的 `/v1/workflow/stream_run` API，传递图像 URL 和其他参数。
* **调度任务状态**：获取返回的 `debug_url`，通过该 URL 定期轮询任务的状态。如果任务完成，就显示生成的提示词。
* **更新 UI**：当任务完成并且结果返回时，右侧的 `generatedPrompt` 区域会显示生成的提示词。
